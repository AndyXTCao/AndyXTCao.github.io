### 题目
删除链表中等于给定值 val 的所有节点。
**示例**：
```
输入: 1->2->6->3->4->5->6, val = 6
输出: 1->2->3->4->5

输入: 1->1 val = 1
输出: nil
```

### 解法
```swift
class ListNode {
    public var val: Int
    public var next: ListNode?
    public init(_ val: Int) {
        self.val = val
        self.next = nil
     }
 }
func removeElements(_ head: ListNode?, _ val: Int) -> ListNode? {
    
    if head == nil {
        return head
    }
    var newHead = head
    //删除值相同的头结点后，可能新的头结点也值相等，用循环解决
    while newHead?.val == val {
        newHead = newHead?.next
    }
    var cur = newHead
    // 确保当前结点后还有结点
    while cur?.next != nil {
        let next = cur?.next
        if next?.val == val {
            cur?.next = next?.next
        }
        else {
            cur = next
        }
    }
    return newHead
}
```
